// LOAD LOGS window.loadLogs = async function () { const { data: userData } = await sb.auth.getUser(); const user = userData.user; const { data } = await sb .from("logs") .select("*") .eq("user_id", user.id); const list = document.getElementById("logList"); const totalDisplay = document.getElementById("totalHours"); list.innerHTML = ""; let total = 0; if (data) { data.forEach(log => { const li = document.createElement("li"); li.textContent = `${log.date} - ${log.aircraft} - ${log.hours} hrs `; const editBtn = document.createElement("button"); editBtn.textContent = "Edit"; editBtn.className = "edit-btn"; editBtn.onclick = () => window.editLog(log); const deleteBtn = document.createElement("button"); deleteBtn.textContent = "Delete"; deleteBtn.className = "delete-btn"; deleteBtn.onclick = () => window.deleteLog(log.id); li.appendChild(editBtn); li.appendChild(deleteBtn); list.appendChild(li); const hoursNum = parseFloat(log.hours); if (!isNaN(hoursNum)) total += hoursNum; }); } totalDisplay.innerText = total.toFixed(1); }; // ADD LOG window.addLog = async function () { const date = document.getElementById("date").value; const aircraft = document.getElementById("aircraft").value; const hours = document.getElementById("hours").value; const { data: userData } = await sb.auth.getUser(); const user = userData.user; const { error } = await sb .from("logs") .insert([{ date, aircraft, hours, user_id: user.id }]); if (error) { document.getElementById("status").innerText = error.message; } else { document.getElementById("status").innerText = "Flight saved!"; window.loadLogs(); } }; // DELETE LOG window.deleteLog = async function (id) { const { error } = await sb .from("logs") .delete() .eq("id", id); if (error) { alert(error.message); } else { window.loadLogs(); } }; // EDIT LOG window.editLog = function (log) { const newDate = prompt("Edit date:", log.date); const newAircraft = prompt("Edit aircraft:", log.aircraft); const newHours = prompt("Edit hours:", log.hours); if (!newDate || !newAircraft || !newHours) return; window.updateLog(log.id, newDate, newAircraft, newHours); }; // UPDATE LOG window.updateLog = async function (id, date, aircraft, hours) { const { error } = await sb .from("logs") .update({ date, aircraft, hours }) .eq("id", id); if (error) { alert(error.message); } else { window.loadLogs(); } }; window.loadLogs(); });
